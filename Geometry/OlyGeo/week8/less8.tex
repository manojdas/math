\begin{lemma}
    Let $ABC$ be a triangle and consider a point $P$ inside the triangle such that $BC$ is tangent to the circumcircles of triangles $ABP$ and $ACP$. Then $M$, the intersection of $BC$ and $AP$, is the midpoint of $BC$.    
\end{lemma}

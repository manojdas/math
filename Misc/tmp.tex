\begin{center}
    \begin{asy}
        unitsize (1 cm);
        real x = 0.45, y = 1;
        for (int i = 0; i < 31; ++i) {
            int v, h;
            if (i % 2 == 0) {
                v = 16 + i # 2;
                h = 1;
            } else if ((i - 1) % 4 == 0) {
                v = 8 + (i - 1) # 4;
                h = 2;
            } else if ((i - 3) % 8 == 0) {
                v = 4 + (i - 3) # 8;
                h = 3;
            } else if ((i - 7) % 16 == 0) {
                v = 2 + (i - 7) # 16;
                h = 4;
            } else {
                v = 1;
                h = 5;
            }
            draw ((i*x, y*h)--(i*x, 0));
            label ("$" + (string)v + "$", (i*x, y*h + 0.2));
        }
    \end{asy}
\end{center}
